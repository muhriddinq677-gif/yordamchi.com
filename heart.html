CSS 
.heart-box {
    position: relative;
    height: 250px;
    background: black;
    overflow: hidden;
}

.heart-box canvas {
    width: 100%;
    height: 100%;
}

.welcome-text {
    position: absolute;
    top: 50%;
    left: 50%;
    transform: translate(-50%, -50%);
    color: white;
    opacity: 0;
}
JS 
function heartX(k){return 15*Math.pow(Math.sin(k),3);}
function heartY(k){return -(13*Math.cos(k)-5*Math.cos(2*k)-2*Math.cos(3*k)-Math.cos(4*k));}

document.querySelectorAll(".heartCanvas").forEach(canvas => {
    const ctx = canvas.getContext("2d");
    const box = canvas.parentElement;
    const text = box.querySelector(".welcome-text");

    canvas.width = box.clientWidth;
    canvas.height = box.clientHeight;

    let t=0,particles=[],done=false,start=null;

    class P{
        constructor(){
            this.x=canvas.width/2;
            this.y=canvas.height/2;
            let a=Math.random()*6.28,s=Math.random()*3;
            this.sx=Math.cos(a)*s;
            this.sy=Math.sin(a)*s;
            this.a=1;
        }
        u(){this.x+=this.sx;this.y+=this.sy;this.a-=0.02;}
        d(){
            ctx.globalAlpha=this.a;
            ctx.fillStyle="#f73487";
            ctx.beginPath();
            ctx.arc(this.x,this.y,2,0,6.28);
            ctx.fill();
            ctx.globalAlpha=1;
        }
    }

    function anim(time){
        ctx.fillStyle="rgba(0,0,0,0.2)";
        ctx.fillRect(0,0,canvas.width,canvas.height);

        if(t<6.28){
            ctx.strokeStyle="#f73487";
            ctx.beginPath();
            for(let j=0;j<=t;j+=0.1){
                let x=canvas.width/2+heartX(j)*5;
                let y=canvas.height/2+heartY(j)*5;
                j?ctx.lineTo(x,y):ctx.moveTo(x,y);
            }
            ctx.stroke();
            t+=0.05;
            if(t>=6.28){done=true;start=time;}
        }

        if(done){
            if(time-start<2000) particles.push(new P());
            else text.style.opacity=1;
        }

        particles.forEach((p,i)=>{
            p.u();p.d();
            if(p.a<=0) particles.splice(i,1);
        });

        requestAnimationFrame(anim);
    }

    anim();
});
